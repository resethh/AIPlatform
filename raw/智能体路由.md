## 一、路由总览

```
                          消息进入
                            │
                            ▼
                ┌───────────────────────┐
                │    路由引擎 (Router)   │
                │                       │
                │  ┌─────────────────┐  │
                │  │ P0 指令路由     │  │  /agent data-analyst ...
                │  │ (显式指定)      │  │
                │  └────────┬────────┘  │
                │           │ 未命中     │
                │  ┌────────▼────────┐  │
                │  │ P1 Session 延续 │  │  对话还没结束，继续上一个 Agent
                │  │ (上下文保持)    │  │
                │  └────────┬────────┘  │
                │           │ 无活跃     │
                │  ┌────────▼────────┐  │
                │  │ P2 @mention 匹配│  │  @数据助手 → data-analyst
                │  │ (精确意图)      │  │
                │  └────────┬────────┘  │
                │           │ 未提及     │
                │  ┌────────▼────────┐  │
                │  │ P3 绑定规则     │  │  这个群/渠道/用户 → 固定 Agent
                │  │ (静态绑定)      │  │
                │  └────────┬────────┘  │
                │           │ 无绑定     │
                │  ┌────────▼────────┐  │
                │  │ P4 意图分类     │  │  LLM 轻量判断："这条消息该给谁？"
                │  │ (智能路由)      │  │
                │  └────────┬────────┘  │
                │           │ 低置信度   │
                │  ┌────────▼────────┐  │
                │  │ P5 关键词匹配   │  │  "销售数据" → data-analyst
                │  │ (规则兜底)      │  │
                │  └────────┬────────┘  │
                │           │ 未命中     │
                │  ┌────────▼────────┐  │
                │  │ P6 默认 Agent   │  │  全局兜底，或提示用户选择
                │  │ (兜底策略)      │  │
                │  └─────────────────┘  │
                └───────────────────────┘
```

**核心原则：每一层只在上一层未命中时才执行，命中即停。**

## 二、路由规则数据模型

### 2.1 数据库表

```sql
-- ============ 路由规则表 ============
CREATE TABLE routing_rules (
    rule_id         SERIAL PRIMARY KEY,
    rule_name       VARCHAR(128) NOT NULL,
    description     TEXT,
    
    -- 匹配条件（AND 语义：所有非空条件都必须满足）
    match_type      VARCHAR(32) NOT NULL,
    -- 可选值: command / session / mention / binding / intent / keyword / default
    
    -- 渠道条件
    channel         VARCHAR(32),              -- dingtalk / wecom / feishu / * (任意)
    chat_type       VARCHAR(16),              -- direct / group / *
    chat_id         VARCHAR(256),             -- 精确群 ID
    chat_id_pattern VARCHAR(256),             -- 群 ID 正则
    
    -- 用户条件
    sender_id       VARCHAR(128),             -- 精确用户
    sender_role     VARCHAR(32),              -- 用户角色: admin / manager / staff
    sender_dept     VARCHAR(128),             -- 用户部门
    tenant_id       VARCHAR(64),              -- 租户/企业 ID
    
    -- 内容条件
    mention_pattern JSONB,                    -- ["@数据助手", "@data-bot"]
    keyword_pattern JSONB,                    -- ["销售数据", "报表", "分析"]
    keyword_mode    VARCHAR(16) DEFAULT 'any',-- any / all / regex
    intent_labels   JSONB,                    -- ["data_analysis", "report_gen"]
    command_prefix  VARCHAR(32),              -- "/agent" 指令前缀
    
    -- 路由目标
    agent_id        VARCHAR(64) NOT NULL REFERENCES agents(agent_id),
    
    -- 优先级和控制
    priority_level  INT NOT NULL,             -- P0-P6 大级别
    priority_order  INT DEFAULT 0,            -- 同级别内的排序（越大越优先）
    enabled         BOOLEAN DEFAULT TRUE,
    
    -- 时间条件（可选）
    active_from     TIME,                     -- 生效开始时间 (09:00)
    active_until    TIME,                     -- 生效结束时间 (18:00)
    active_weekdays INT[],                    -- 生效星期 [1,2,3,4,5] = 工作日
    
    -- 元数据
    created_by      VARCHAR(64),
    created_at      TIMESTAMPTZ DEFAULT NOW(),
    updated_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_routing_enabled ON routing_rules(enabled, priority_level, priority_order DESC);
CREATE INDEX idx_routing_channel ON routing_rules(channel, chat_type);
CREATE INDEX idx_routing_agent ON routing_rules(agent_id);


-- ============ 路由日志表（追溯为什么选了这个 Agent）============
CREATE TABLE routing_logs (
    id              BIGSERIAL PRIMARY KEY,
    trace_id        VARCHAR(64) NOT NULL,
    message_id      VARCHAR(128) NOT NULL,
    
    -- 路由过程
    evaluated_rules JSONB NOT NULL,           -- 评估了哪些规则，每条的匹配结果
    matched_rule_id INT,                      -- 最终命中的规则 ID
    matched_level   VARCHAR(16),              -- 命中的优先级层 P0-P6
    agent_id        VARCHAR(64),              -- 最终路由到的 Agent
    confidence      FLOAT,                    -- 置信度 (仅 intent 路由有)
    
    -- 耗时
    routing_ms      INT,                      -- 路由决策耗时
    
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_routing_logs_trace ON routing_logs(trace_id);
CREATE INDEX idx_routing_logs_time ON routing_logs(created_at);
```

## 三、路由引擎实现

```python
# pipeline/agent_router.py

from dataclasses import dataclass, field
from typing import Optional, List, Dict
from datetime import datetime, time as dtime
import re, json, logging

log = logging.getLogger(__name__)


@dataclass
class RouteResult:
    """路由结果"""
    agent_id: str
    rule_id: Optional[int] = None
    rule_name: str = ""
    matched_level: str = ""           # P0-P6
    confidence: float = 1.0           # 1.0 = 确定, <1.0 = 意图推断
    reason: str = ""                  # 人类可读的路由原因
    evaluated: List[dict] = field(default_factory=list)  # 评估过程
    routing_ms: int = 0


class AgentRouter:
    """
    多层级路由引擎
    
    设计参考:
    - OpenClaw binding 机制 (peer > channel > default)
    - Nginx location 匹配 (最具体优先)
    - 增加: 意图分类 (LLM 轻量推断) + 时间条件
    """
    
    def __init__(self, redis_client, db_pool, intent_classifier=None):
        self.redis = redis_client
        self.db = db_pool
        self.intent_classifier = intent_classifier  # 可选的 LLM 意图分类器
        
        # 规则缓存（内存，定期刷新）
        self._rules_cache: List[dict] = []
        self._rules_cache_ts: float = 0
        self._cache_ttl: int = 60  # 60 秒刷新一次
    
    
    async def resolve(self, message, session) -> RouteResult:
        """
        主路由入口：按优先级依次尝试每一层
        """
        start_ts = datetime.now()
        evaluated = []
        
        # 加载规则
        rules = await self._load_rules()
        
        # ══════ P0: 指令路由 ══════
        result = self._match_command(message)
        evaluated.append({"level": "P0_command", "matched": result is not None})
        if result:
            result.evaluated = evaluated
            result.routing_ms = self._elapsed_ms(start_ts)
            return result
        
        # ══════ P1: Session 延续 ══════
        result = await self._match_session(message, session)
        evaluated.append({"level": "P1_session", "matched": result is not None})
        if result:
            result.evaluated = evaluated
            result.routing_ms = self._elapsed_ms(start_ts)
            return result
        
        # ══════ P2: @mention ══════
        result = self._match_mention(message, rules)
        evaluated.append({"level": "P2_mention", "matched": result is not None})
        if result:
            result.evaluated = evaluated
            result.routing_ms = self._elapsed_ms(start_ts)
            return result
        
        # ══════ P3: 绑定规则 ══════
        result = self._match_binding(message, rules)
        evaluated.append({"level": "P3_binding", "matched": result is not None})
        if result:
            result.evaluated = evaluated
            result.routing_ms = self._elapsed_ms(start_ts)
            return result
        
        # ══════ P4: 意图分类（LLM 推断）══════
        if self.intent_classifier:
            result = await self._match_intent(message, rules)
            evaluated.append({
                "level": "P4_intent", 
                "matched": result is not None,
                "confidence": result.confidence if result else 0
            })
            if result:
                result.evaluated = evaluated
                result.routing_ms = self._elapsed_ms(start_ts)
                return result
        
        # ══════ P5: 关键词匹配 ══════
        result = self._match_keyword(message, rules)
        evaluated.append({"level": "P5_keyword", "matched": result is not None})
        if result:
            result.evaluated = evaluated
            result.routing_ms = self._elapsed_ms(start_ts)
            return result
        
        # ══════ P6: 默认 ══════
        result = self._match_default(message, rules)
        evaluated.append({"level": "P6_default", "matched": result is not None})
        if result:
            result.evaluated = evaluated
            result.routing_ms = self._elapsed_ms(start_ts)
            return result
        
        # 全部未命中（不应该发生，除非没配 default）
        return RouteResult(
            agent_id="",
            matched_level="NONE",
            reason="所有路由规则均未命中",
            evaluated=evaluated,
            routing_ms=self._elapsed_ms(start_ts),
        )

    # ─────────────────────────────────────────────
    #  P0: 指令路由
    # ─────────────────────────────────────────────
    def _match_command(self, message) -> Optional[RouteResult]:
        """
        识别显式指令：
          /agent data-analyst 帮我分析数据
          /ask code-reviewer review this PR
          /switch finance-assistant
        """
        text = (message.content.text or "").strip()
        
        command_prefixes = ["/agent ", "/ask ", "/switch "]
        
        for prefix in command_prefixes:
            if text.lower().startswith(prefix):
                remainder = text[len(prefix):].strip()
                parts = remainder.split(None, 1)  # 分割：agent_id + 剩余内容
                
                if parts:
                    target_agent = parts[0]
                    # 重写消息内容（去掉指令部分）
                    if len(parts) > 1:
                        message.content.text = parts[1]
                    else:
                        message.content.text = ""
                    
                    return RouteResult(
                        agent_id=target_agent,
                        matched_level="P0",
                        reason=f"用户指令: {prefix.strip()} → {target_agent}",
                    )
        
        return None

    # ─────────────────────────────────────────────
    #  P1: Session 延续
    # ─────────────────────────────────────────────
    async def _match_session(self, message, session) -> Optional[RouteResult]:
        """
        如果当前 Session 还活跃且绑定了 Agent，继续沿用
        
        条件：
        1. Session 存在且 status=active
        2. Agent 仍然可用
        3. Session 未过期（idle 超时检查）
        4. 用户没有发送"换人"/"重新开始"等切换信号
        """
        if not session or not session.agent_id:
            return None
        
        if session.status != "active":
            return None
        
        # 检查切换信号
        switch_signals = ["换个助手", "重新开始", "换人", "/new", "/reset", "/switch"]
        text = (message.content.text or "").strip()
        if any(signal in text for signal in switch_signals):
            # 标记 Session 需要重置
            session.status = "expired"
            await self._save_session_status(session)
            return None
        
        # 检查 idle 超时（比如 30 分钟无交互就不延续）
        idle_seconds = (datetime.now() - session.updated_at).total_seconds()
        idle_limit = await self._get_idle_limit(session.agent_id)
        if idle_seconds > idle_limit:
            return None
        
        # 检查 Agent 是否仍可用
        agent = await self._get_agent_status(session.agent_id)
        if not agent or agent.get("status") != "active":
            return None
        
        return RouteResult(
            agent_id=session.agent_id,
            matched_level="P1",
            reason=f"Session 延续: {session.session_id} → {session.agent_id} "
                   f"(空闲 {int(idle_seconds)}s)",
        )

    # ─────────────────────────────────────────────
    #  P2: @mention 匹配
    # ─────────────────────────────────────────────
    def _match_mention(self, message, rules: list) -> Optional[RouteResult]:
        """
        匹配消息中的 @提及
        
        三种来源：
        1. 渠道原生 mention（飞书/钉钉的结构化 mention 数据）
        2. 文本中的 @xxx 模式
        3. Agent 的 mention_pattern 配置
        """
        mentions = set()
        
        # 来源 1: 渠道原生 mention
        if message.content.mentions:
            for m in message.content.mentions:
                mentions.add(m.lower().strip().lstrip("@"))
        
        # 来源 2: 从文本中提取 @xxx
        text = message.content.text or ""
        text_mentions = re.findall(r"@(\S+)", text)
        for m in text_mentions:
            mentions.add(m.lower().strip())
        
        if not mentions:
            return None
        
        # 匹配规则
        mention_rules = [r for r in rules 
                         if r["match_type"] == "mention" 
                         and r["enabled"]
                         and self._check_channel(r, message)]
        
        # 按 priority_order 降序排列
        mention_rules.sort(key=lambda r: r["priority_order"], reverse=True)
        
        for rule in mention_rules:
            patterns = rule.get("mention_pattern") or []
            for pattern in patterns:
                clean_pattern = pattern.lower().strip().lstrip("@")
                if clean_pattern in mentions:
                    # 命中！清理消息文本中的 @mention
                    cleaned_text = re.sub(
                        rf"@{re.escape(pattern)}\s*", "", text, flags=re.IGNORECASE
                    ).strip()
                    message.content.text = cleaned_text
                    
                    return RouteResult(
                        agent_id=rule["agent_id"],
                        rule_id=rule["rule_id"],
                        rule_name=rule["rule_name"],
                        matched_level="P2",
                        reason=f"@mention: '{pattern}' → {rule['agent_id']}",
                    )
        
        return None

    # ─────────────────────────────────────────────
    #  P3: 绑定规则（静态路由）
    # ─────────────────────────────────────────────
    def _match_binding(self, message, rules: list) -> Optional[RouteResult]:
        """
        静态绑定规则，支持多维度组合匹配（AND 语义）
        
        匹配维度（具体度从高到低）:
        1. channel + chat_id + sender_id    （最精确：某个群里的某个人）
        2. channel + chat_id                （群绑定）
        3. channel + sender_id              （用户绑定）
        4. channel + sender_role            （角色绑定）
        5. channel + sender_dept            （部门绑定）
        6. channel + chat_type              （渠道+类型绑定）
        7. tenant_id                        （租户级绑定）
        
        同时检查时间条件（如果配置了）
        """
        binding_rules = [r for r in rules 
                         if r["match_type"] == "binding" 
                         and r["enabled"]]
        
        # 按 priority_order 降序 → 同 order 按具体度降序
        binding_rules.sort(key=lambda r: (
            r["priority_order"],
            self._calc_specificity(r)
        ), reverse=True)
        
        for rule in binding_rules:
            if self._binding_matches(rule, message):
                return RouteResult(
                    agent_id=rule["agent_id"],
                    rule_id=rule["rule_id"],
                    rule_name=rule["rule_name"],
                    matched_level="P3",
                    reason=self._describe_binding_match(rule),
                )
        
        return None
    

    def _binding_matches(self, rule: dict, message) -> bool:
        """检查一条绑定规则是否匹配（所有非空条件都要满足）"""
        src = message.source
        
        # 渠道
        if rule.get("channel") and rule["channel"] != "*":
            if rule["channel"] != src.channel:
                return False
        
        # 聊天类型
        if rule.get("chat_type") and rule["chat_type"] != "*":
            if rule["chat_type"] != src.chat_type:
                return False
        
        # 精确群 ID
        if rule.get("chat_id"):
            if rule["chat_id"] != src.chat_id:
                return False
        
        # 群 ID 正则
        if rule.get("chat_id_pattern"):
            if not re.match(rule["chat_id_pattern"], src.chat_id or ""):
                return False
        
        # 精确用户
        if rule.get("sender_id"):
            if rule["sender_id"] != src.sender_id:
                return False
        
        # 用户角色（需要从用户信息中获取）
        if rule.get("sender_role"):
            user_role = getattr(message.metadata, "user_role", None)
            if user_role != rule["sender_role"]:
                return False
        
        # 用户部门
        if rule.get("sender_dept"):
            user_dept = getattr(message.metadata, "user_dept", None)
            if user_dept != rule["sender_dept"]:
                return False
        
        # 租户
        if rule.get("tenant_id"):
            if rule["tenant_id"] != src.tenant_id:
                return False
        
        # 时间条件
        if not self._check_time_condition(rule):
            return False
        
        return True
    

    def _calc_specificity(self, rule: dict) -> int:
        """计算规则具体度（维度越多越具体，越优先）"""
        score = 0
        if rule.get("chat_id"):      score += 100
        if rule.get("sender_id"):    score += 80
        if rule.get("sender_role"):  score += 40
        if rule.get("sender_dept"):  score += 30
        if rule.get("chat_type") and rule["chat_type"] != "*": score += 20
        if rule.get("channel") and rule["channel"] != "*":     score += 10
        if rule.get("tenant_id"):    score += 5
        return score
    

    def _check_time_condition(self, rule: dict) -> bool:
        """检查时间条件"""
        now = datetime.now()
        
        # 星期检查
        weekdays = rule.get("active_weekdays")
        if weekdays:
            # Python: Monday=0, 但我们用 1=Monday
            current_weekday = now.isoweekday()  # 1=Mon ... 7=Sun
            if current_weekday not in weekdays:
                return False
        
        # 时间段检查
        active_from = rule.get("active_from")
        active_until = rule.get("active_until")
        
        if active_from and active_until:
            from_time = dtime.fromisoformat(active_from)
            until_time = dtime.fromisoformat(active_until)
            current_time = now.time()
            
            if from_time <= until_time:
                # 正常时间段 (09:00 - 18:00)
                if not (from_time <= current_time <= until_time):
                    return False
            else:
                # 跨午夜 (22:00 - 06:00)
                if until_time < current_time < from_time:
                    return False
        
        return True

    # ─────────────────────────────────────────────
    #  P4: 意图分类（LLM 轻量推断）
    # ─────────────────────────────────────────────
    async def _match_intent(self, message, rules: list) -> Optional[RouteResult]:
        """
        用 LLM 做轻量意图分类
        
        设计要点：
        1. 用最便宜最快的模型（qwen-turbo / gpt-4o-mini）
        2. 严格限制 token（prompt < 500 token, 响应 < 50 token）
        3. 结果缓存（相似问题不重复调用）
        4. 超时 fallback（2 秒超时就跳过）
        5. 返回置信度，低于阈值不采用
        """
        text = message.content.text or ""
        if len(text) < 5:
            return None  # 太短不做意图分类
        
        # 收集所有意图规则
        intent_rules = [r for r in rules 
                        if r["match_type"] == "intent" 
                        and r["enabled"]]
        if not intent_rules:
            return None
        
        # 构建 Agent 目录（给 LLM 做选择）
        agent_catalog = []
        label_to_agent = {}
        for rule in intent_rules:
            labels = rule.get("intent_labels") or []
            for label in labels:
                label_to_agent[label] = rule
                agent_catalog.append({
                    "label": label,
                    "agent": rule["agent_id"],
                    "description": rule.get("description", ""),
                })
        
        # 先查缓存
        cache_key = f"intent_cache:{self._text_hash(text)}"
        cached = await self.redis.get(cache_key)
        if cached:
            cached_data = json.loads(cached)
            rule = label_to_agent.get(cached_data["label"])
            if rule:
                return RouteResult(
                    agent_id=rule["agent_id"],
                    rule_id=rule["rule_id"],
                    rule_name=rule["rule_name"],
                    matched_level="P4",
                    confidence=cached_data["confidence"],
                    reason=f"意图分类(缓存): {cached_data['label']} "
                           f"→ {rule['agent_id']} (置信度 {cached_data['confidence']:.0%})",
                )
        
        # 调 LLM 做意图分类
        try:
            result = await asyncio.wait_for(
                self.intent_classifier.classify(text, agent_catalog),
                timeout=2.0  # 最多 2 秒
            )
        except asyncio.TimeoutError:
            log.warning(f"意图分类超时, 跳过 P4")
            return None
        except Exception as e:
            log.error(f"意图分类异常: {e}")
            return None
        
        if not result or result.confidence < 0.7:  # 置信度阈值
            return None
        
        # 写缓存（5 分钟）
        await self.redis.set(cache_key, json.dumps({
            "label": result.label,
            "confidence": result.confidence,
        }), ex=300)
        
        rule = label_to_agent.get(result.label)
        if not rule:
            return None
        
        return RouteResult(
            agent_id=rule["agent_id"],
            rule_id=rule["rule_id"],
            rule_name=rule["rule_name"],
            matched_level="P4",
            confidence=result.confidence,
            reason=f"意图分类: '{result.label}' → {rule['agent_id']} "
                   f"(置信度 {result.confidence:.0%})",
        )

    # ─────────────────────────────────────────────
    #  P5: 关键词匹配
    # ─────────────────────────────────────────────
    def _match_keyword(self, message, rules: list) -> Optional[RouteResult]:
        """
        关键词匹配，支持三种模式：
        - any: 包含任一关键词即命中
        - all: 包含所有关键词才命中
        - regex: 正则表达式匹配
        """
        text = (message.content.text or "").lower()
        if not text:
            return None
        
        keyword_rules = [r for r in rules 
                         if r["match_type"] == "keyword" 
                         and r["enabled"]
                         and self._check_channel(r, message)]
        
        keyword_rules.sort(key=lambda r: r["priority_order"], reverse=True)
        
        best_match = None
        best_hit_count = 0
        
        for rule in keyword_rules:
            patterns = rule.get("keyword_pattern") or []
            mode = rule.get("keyword_mode", "any")
            
            if mode == "regex":
                for pattern in patterns:
                    if re.search(pattern, text, re.IGNORECASE):
                        return RouteResult(
                            agent_id=rule["agent_id"],
                            rule_id=rule["rule_id"],
                            rule_name=rule["rule_name"],
                            matched_level="P5",
                            reason=f"关键词(正则): /{pattern}/ → {rule['agent_id']}",
                        )
            
            elif mode == "all":
                if all(kw.lower() in text for kw in patterns):
                    return RouteResult(
                        agent_id=rule["agent_id"],
                        rule_id=rule["rule_id"],
                        rule_name=rule["rule_name"],
                        matched_level="P5",
                        reason=f"关键词(全匹配): {patterns} → {rule['agent_id']}",
                    )
            
            else:  # mode == "any"
                hit_count = sum(1 for kw in patterns if kw.lower() in text)
                if hit_count > 0 and hit_count > best_hit_count:
                    best_hit_count = hit_count
                    best_match = RouteResult(
                        agent_id=rule["agent_id"],
                        rule_id=rule["rule_id"],
                        rule_name=rule["rule_name"],
                        matched_level="P5",
                        confidence=hit_count / len(patterns),  # 命中比例作为置信度
                        reason=f"关键词({hit_count}/{len(patterns)}命中): "
                               f"→ {rule['agent_id']}",
                    )
        
        return best_match

    # ─────────────────────────────────────────────
    #  P6: 默认路由
    # ─────────────────────────────────────────────
    def _match_default(self, message, rules: list) -> Optional[RouteResult]:
        """
        兜底策略，两种模式：
        1. 直接路由到默认 Agent
        2. 返回一个"选择器 Agent"，让它引导用户选择
        """
        default_rules = [r for r in rules 
                         if r["match_type"] == "default" 
                         and r["enabled"]
                         and self._check_channel(r, message)]
        
        if default_rules:
            rule = default_rules[0]
            return RouteResult(
                agent_id=rule["agent_id"],
                rule_id=rule["rule_id"],
                rule_name=rule["rule_name"],
                matched_level="P6",
                reason=f"默认路由 → {rule['agent_id']}",
            )
        
        # 硬兜底
        return RouteResult(
            agent_id="general-assistant",
            matched_level="P6",
            reason="硬编码默认 → general-assistant",
        )

    # ─────────────────────────────────────────────
    #  辅助方法
    # ─────────────────────────────────────────────
    def _check_channel(self, rule: dict, message) -> bool:
        """检查规则的渠道条件是否匹配"""
        if not rule.get("channel") or rule["channel"] == "*":
            return True
        return rule["channel"] == message.source.channel
    

    def _describe_binding_match(self, rule: dict) -> str:
        """生成人类可读的匹配描述"""
        parts = []
        if rule.get("channel") and rule["channel"] != "*":
            parts.append(f"渠道={rule['channel']}")
        if rule.get("chat_id"):
            parts.append(f"群={rule['chat_id'][:20]}...")
        if rule.get("sender_id"):
            parts.append(f"用户={rule['sender_id'][:20]}...")
        if rule.get("sender_role"):
            parts.append(f"角色={rule['sender_role']}")
        if rule.get("sender_dept"):
            parts.append(f"部门={rule['sender_dept']}")
        if rule.get("active_from"):
            parts.append(f"时间={rule['active_from']}-{rule.get('active_until','')}")
        
        desc = " + ".join(parts) if parts else "默认绑定"
        return f"绑定规则[{rule['rule_name']}]: {desc} → {rule['agent_id']}"
    

    def _text_hash(self, text: str) -> str:
        """生成文本摘要 hash（意图缓存用）"""
        import hashlib
        normalized = text.strip().lower()[:200]
        return hashlib.md5(normalized.encode()).hexdigest()[:16]
    

    def _elapsed_ms(self, start: datetime) -> int:
        return int((datetime.now() - start).total_seconds() * 1000)
    

    async def _load_rules(self) -> list:
        """加载路由规则（带缓存）"""
        import time
        now = time.time()
        
        if now - self._rules_cache_ts < self._cache_ttl and self._rules_cache:
            return self._rules_cache
        
        # 从 Redis 缓存读
        cached = await self.redis.get("routing:rules:all")
        if cached:
            self._rules_cache = json.loads(cached)
            self._rules_cache_ts = now
            return self._rules_cache
        
        # 从 DB 读
        async with self.db.acquire() as conn:
            rows = await conn.fetch("""
                SELECT * FROM routing_rules 
                WHERE enabled = TRUE 
                ORDER BY priority_level ASC, priority_order DESC
            """)
            rules = [dict(row) for row in rows]
        
        # 写 Redis 缓存
        await self.redis.set("routing:rules:all", json.dumps(rules, default=str), ex=60)
        self._rules_cache = rules
        self._rules_cache_ts = now
        
        return rules
    
    
    async def _get_agent_status(self, agent_id: str) -> Optional[dict]:
        """获取 Agent 状态"""
        return await self.redis.hgetall(f"agent:{agent_id}:config")
    
    
    async def _get_idle_limit(self, agent_id: str) -> int:
        """获取 Agent 的 Session 空闲超时（秒）"""
        config = await self._get_agent_status(agent_id)
        return int(config.get("session_idle_seconds", 1800))  # 默认 30 分钟
```

## 四、意图分类器实现

```python
# pipeline/intent_classifier.py

from dataclasses import dataclass
from typing import List, Optional
import json, httpx

@dataclass
class IntentResult:
    label: str
    confidence: float
    reasoning: str = ""


class IntentClassifier:
    """
    用最小成本做意图分类
    
    设计原则:
    1. 用最便宜的模型 (qwen-turbo ~0.001元/千token)
    2. Prompt < 500 token, 响应 < 50 token
    3. 结构化输出 (JSON mode)
    4. 不做复杂推理，只做分类
    """
    
    def __init__(self, llm_client, model: str = "qwen-turbo"):
        self.llm = llm_client
        self.model = model
    
    async def classify(self, text: str, 
                       agent_catalog: List[dict]) -> Optional[IntentResult]:
        """
        对用户消息做意图分类
        
        Args:
            text: 用户消息
            agent_catalog: 可选 Agent 列表 [{label, agent, description}]
        
        Returns:
            IntentResult or None
        """
        # 构建分类 prompt
        catalog_desc = "\n".join([
            f"- {item['label']}: {item.get('description', item['agent'])}"
            for item in agent_catalog
        ])
        
        prompt = f"""你是一个消息分类器。根据用户消息，从下面的类别中选择最匹配的一个。

可选类别:
{catalog_desc}
- none: 以上都不匹配

用户消息: "{text[:200]}"

返回 JSON: {{"label": "类别名", "confidence": 0.0到1.0}}
只返回 JSON，不要其他内容。"""
        
        response = await self.llm.chat_completion(
            model=self.model,
            messages=[{"role": "user", "content": prompt}],
            temperature=0,
            max_tokens=50,
            response_format={"type": "json_object"},
        )
        
        try:
            result = json.loads(response.choices[0].message.content)
            label = result.get("label", "none")
            confidence = float(result.get("confidence", 0))
            
            if label == "none" or confidence < 0.5:
                return None
            
            return IntentResult(
                label=label,
                confidence=confidence,
            )
        except (json.JSONDecodeError, KeyError, ValueError):
            return None
```

