```python
# adapters/dingtalk_adapter.py

from fastapi import APIRouter, Request, HTTPException
import hashlib, hmac, base64, time, json

router = APIRouter(prefix="/webhook/dingtalk")

@router.post("/callback")
async def dingtalk_callback(request: Request):
    body = await request.body()
    headers = request.headers
    
    # 1. 验签
    timestamp = headers.get("timestamp")
    sign = headers.get("sign")
    if not verify_dingtalk_sign(timestamp, sign, DINGTALK_APP_SECRET):
        raise HTTPException(403, "签名验证失败")
    
    data = json.loads(body)
    
    # 2. 转换为统一消息格式
    message = UnifiedMessage(
        message_id=f"msg_dingtalk_{data['msgId']}",
        trace_id=generate_trace_id(),
        source=MessageSource(
            channel="dingtalk",
            app_id=data.get("chatbotCorpId"),
            chat_id=data.get("conversationId"),
            chat_type="group" if data.get("conversationType") == "2" else "direct",
            sender_id=data["senderStaffId"],
            sender_name=data.get("senderNick", ""),
            tenant_id=data.get("chatbotCorpId"),
            message_id_raw=data["msgId"],
        ),
        content=parse_dingtalk_content(data),
        metadata=MessageMetadata(
            received_at=datetime.now(tz=timezone(timedelta(hours=8))),
            enterprise_id=data.get("chatbotCorpId"),
        ),
    )
    
    # 3. 发送到处理管道
    await message_pipeline.process(message)
    
    # 4. 钉钉要求立即返回（异步处理）
    return {"msgtype": "empty"}


def verify_dingtalk_sign(timestamp: str, sign: str, secret: str) -> bool:
    """钉钉签名验证"""
    string_to_sign = f"{timestamp}\n{secret}"
    hmac_code = hmac.new(
        secret.encode("utf-8"),
        string_to_sign.encode("utf-8"),
        digestmod=hashlib.sha256
    ).digest()
    expected = base64.b64encode(hmac_code).decode("utf-8")
    return sign == expected
```

### 3.2 消息处理管道 (Message Pipeline)

适配器转换完后，消息进入统一处理管道：

```python
# pipeline/message_pipeline.py

class MessagePipeline:
    """统一消息处理管道"""
    
    def __init__(self, redis_client, session_store, agent_registry, router):
        self.redis = redis_client
        self.session_store = session_store
        self.agent_registry = agent_registry
        self.router = router
    
    async def process(self, message: UnifiedMessage):
        """
        完整处理流程:
        验证 → 去重 → Session解析 → Agent路由 → 限流 → 入队
        """
        trace_id = message.trace_id
        log.info(f"[{trace_id}] 收到消息: channel={message.source.channel}, "
                 f"sender={message.source.sender_id}")
        
        # ── Step 1: 幂等去重 ──
        dedup_key = f"dedup:{message.source.channel}:{message.source.message_id_raw}"
        if await self.redis.set(dedup_key, "1", nx=True, ex=300) is None:
            log.warn(f"[{trace_id}] 重复消息，跳过")
            return
        
        # ── Step 2: 用户身份认证 ──
        user = await self.authenticate_user(message)
        if not user:
            await self.reply_error(message, "身份验证失败，请联系管理员")
            return
        
        # ── Step 3: 解析/创建 Session ──
        session = await self.session_store.resolve_session(message)
        message.routing.session_id = session.session_id
        
        # ── Step 4: 路由到 Agent ──
        agent = await self.router.resolve_agent(message, session)
        if not agent:
            await self.reply_error(message, "未找到匹配的智能体，请 @指定助手")
            return
        message.routing.agent_id = agent.agent_id
        
        # ── Step 5: 限流检查 ──
        if not await self.check_rate_limit(user, agent):
            await self.reply_error(message, "请求过于频繁，请稍后再试")
            return
        
        # ── Step 6: 入队 ──
        await self.enqueue_task(message, session, agent)
        
        log.info(f"[{trace_id}] 任务入队: agent={agent.agent_id}, "
                 f"session={session.session_id}")
    

    async def authenticate_user(self, message: UnifiedMessage) -> Optional[User]:
        """
        根据渠道 + sender_id 查找企业用户
        可对接企业通讯录 API 或本地用户表
        """
        user_key = f"{message.source.channel}:{message.source.sender_id}"
        user = await self.redis.hgetall(f"user:{user_key}")
        if user:
            return User(**user)
        
        # 回源查询（调渠道 API 获取用户信息，缓存）
        user = await self.fetch_user_from_channel(message)
        if user:
            await self.redis.hset(f"user:{user_key}", mapping=user.dict())
            await self.redis.expire(f"user:{user_key}", 3600)
        return user

    
    async def check_rate_limit(self, user: User, agent: AgentConfig) -> bool:
        """令牌桶限流（per user + per agent）"""
        # 用户级限流: 10 req/min
        user_key = f"ratelimit:user:{user.user_id}"
        user_count = await self.redis.incr(user_key)
        if user_count == 1:
            await self.redis.expire(user_key, 60)
        if user_count > 10:
            return False
        
        # Agent 级限流
        agent_key = f"ratelimit:agent:{agent.agent_id}"
        agent_count = await self.redis.incr(agent_key)
        if agent_count == 1:
            await self.redis.expire(agent_key, 60)
        if agent_count > agent.limits.max_concurrent:
            return False
        
        return True
    

    async def enqueue_task(self, message: UnifiedMessage, 
                           session: Session, agent: AgentConfig):
        """写入 Redis Streams 任务队列"""
        
        priority = message.routing.priority or "normal"
        queue_name = f"task:{priority}"
        
        task_data = {
            "task_id": f"task_{uuid4().hex[:12]}",
            "trace_id": message.trace_id,
            "message_id": message.message_id,
            "agent_id": agent.agent_id,
            "session_id": session.session_id,
            "user_id": message.source.sender_id,
            "channel": message.source.channel,
            "chat_id": message.source.chat_id,
            "chat_type": message.source.chat_type,
            "text": message.content.text,
            "attachments": json.dumps(message.content.attachments or []),
            "model": agent.model_config.primary,
            "timeout": str(agent.limits.timeout_seconds),
            "enqueued_at": datetime.now().isoformat(),
            
            # 回复路由信息（Worker 完成后需要知道往哪回复）
            "reply_channel": message.source.channel,
            "reply_chat_id": message.source.chat_id,
            "reply_message_id": message.source.message_id_raw,
            "reply_app_id": message.source.app_id,
            "reply_tenant_id": message.source.tenant_id,
        }
        
        await self.redis.xadd(queue_name, task_data, maxlen=100000)
```

### 3.3 路由决策层 (Agent Router)

```python
# pipeline/agent_router.py

class AgentRouter:
    """根据消息内容 + 上下文决定路由到哪个 Agent"""
    
    def __init__(self, agent_registry):
        self.registry = agent_registry
    
    async def resolve_agent(self, message: UnifiedMessage, 
                            session: Session) -> Optional[AgentConfig]:
        """
        路由优先级（参考 OpenClaw 的 binding 机制）:
        1. Session 已绑定 Agent（对话中）→ 延续
        2. @mention 匹配
        3. 渠道+群组 绑定规则
        4. 关键词匹配
        5. 默认 Agent
        """
        
        # ── 优先级 1: Session 已有 Agent（对话延续）──
        if session.agent_id and session.status == "active":
            agent = await self.registry.get(session.agent_id)
            if agent and agent.status == "active":
                return agent
        
        # ── 优先级 2: @mention 匹配 ──
        if message.content.mentions:
            for agent in await self.registry.list_active():
                for rule in agent.routing_rules:
                    if rule["match"] == "mentions":
                        for pattern in rule["pattern"]:
                            if pattern in message.content.mentions:
                                return agent
        
        # ── 优先级 3: 渠道+群组 绑定 ──
        for agent in await self.registry.list_active():
            for rule in agent.routing_rules:
                if rule["match"] == "channel":
                    p = rule["pattern"]
                    if (p.get("channel") == message.source.channel and
                        p.get("chat_id") == message.source.chat_id):
                        return agent
        
        # ── 优先级 4: 关键词匹配 ──
        text = message.content.text
        for agent in await self.registry.list_active():
            for rule in agent.routing_rules:
                if rule["match"] == "keyword":
                    if any(kw in text for kw in rule["pattern"]):
                        return agent
        
        # ── 优先级 5: 默认 Agent ──
        default = await self.registry.get_default()
        return default
```

### 3.4 Worker 执行层

```python
# worker/agent_worker.py

class AgentWorker:
    """
    从 Redis Streams 消费任务，执行 Agent Run
    每台机器部署 1 个 Worker 进程，内部跑 N 个并发协程
    """
    
    def __init__(self, worker_id: str, redis_client, 
                 max_concurrent: int = 50):
        self.worker_id = worker_id
        self.redis = redis_client
        self.max_concurrent = max_concurrent
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.active_runs = 0
        
        # LLM 客户端池（多 Key 轮转）
        self.llm_pool = LLMClientPool(config=LLM_CONFIG)
        
        # Session 存储
        self.session_store = SessionStore(redis_client, pg_pool)
        
        # Tool 执行器
        self.tool_executor = ToolExecutor()
    
    
    async def start(self):
        """启动消费循环"""
        # 创建消费者组
        for queue in ["task:high", "task:normal", "task:low"]:
            try:
                await self.redis.xgroup_create(queue, "workers", id="0", mkstream=True)
            except Exception:
                pass  # 已存在
        
        log.info(f"Worker {self.worker_id} 启动, max_concurrent={self.max_concurrent}")
        
        # 启动消费协程
        await asyncio.gather(
            self.consume_loop("task:high", poll_interval=100),     # 高优先级：100ms 检查
            self.consume_loop("task:normal", poll_interval=500),   # 普通：500ms
            self.consume_loop("task:low", poll_interval=2000),     # 低优先级：2s
        )
    
    
    async def consume_loop(self, queue: str, poll_interval: int):
        """持续消费指定队列"""
        while True:
            if self.active_runs >= self.max_concurrent:
                await asyncio.sleep(poll_interval / 1000)
                continue
            
            try:
                messages = await self.redis.xreadgroup(
                    "workers", self.worker_id,
                    {queue: ">"},
                    count=1,
                    block=poll_interval
                )
                
                if messages:
                    for stream, entries in messages:
                        for msg_id, task_data in entries:
                            asyncio.create_task(
                                self.execute_run(task_data, stream, msg_id)
                            )
            except Exception as e:
                log.error(f"消费异常: {e}")
                await asyncio.sleep(1)
    
    
    async def execute_run(self, task: dict, stream: str, msg_id: str):
        """执行一个完整的 Agent Run"""
        async with self.semaphore:
            self.active_runs += 1
            task_id = task["task_id"]
            trace_id = task["trace_id"]
            
            try:
                log.info(f"[{trace_id}] 开始执行: agent={task['agent_id']}")
                
                # ── 1. 获取 Session 锁 ──
                lock = await self.acquire_session_lock(task["session_id"])
                if not lock:
                    # 同一 Session 有其他任务在跑，重新入队
                    await self.redis.xadd(stream, task)
                    await self.redis.xack(stream, "workers", msg_id)
                    return
                
                try:
                    # ── 2. 加载 Agent 配置 ──
                    agent = await self.load_agent_config(task["agent_id"])
                    
                    # ── 3. 加载 Session 上下文 ──
                    session = await self.session_store.load(task["session_id"])
                    
                    # ── 4. 构建 Prompt ──
                    messages = self.build_messages(agent, session, task["text"])
                    
                    # ── 5. 调用 LLM（可能多轮 Tool Calling）──
                    result = await self.run_agent_loop(
                        agent, messages, task, trace_id
                    )
                    
                    # ── 6. 保存 Session ──
                    await self.session_store.append(
                        task["session_id"],
                        user_message=task["text"],
                        assistant_message=result.final_text,
                        usage=result.usage
                    )
                    
                    # ── 7. 发送回复 ──
                    await self.publish_reply(task, result)
                    
                    log.info(f"[{trace_id}] 执行完成: tokens={result.usage.total}")
                    
                finally:
                    await self.release_session_lock(task["session_id"])
                
                # ── 8. ACK 队列 ──
                await self.redis.xack(stream, "workers", msg_id)
                
            except asyncio.TimeoutError:
                log.error(f"[{trace_id}] 执行超时")
                await self.publish_error_reply(task, "处理超时，请稍后重试")
                await self.redis.xack(stream, "workers", msg_id)
                
            except Exception as e:
                log.error(f"[{trace_id}] 执行异常: {e}", exc_info=True)
                await self.publish_error_reply(task, "处理出错，请稍后重试")
                await self.redis.xack(stream, "workers", msg_id)
                
            finally:
                self.active_runs -= 1
    

    async def run_agent_loop(self, agent: AgentConfig, messages: list,
                             task: dict, trace_id: str) -> AgentResult:
        """
        Agent 执行循环（参考 OpenClaw 的 Agent Loop）
        支持多轮 Tool Calling
        """
        max_iterations = 10
        all_tool_results = []
        
        for iteration in range(max_iterations):
            # 调用 LLM
            response = await asyncio.wait_for(
                self.llm_pool.chat_completion(
                    model=agent.model_config.primary,
                    messages=messages,
                    tools=agent.get_tool_definitions(),
                    temperature=agent.model_config.temperature,
                    max_tokens=agent.model_config.max_tokens,
                    stream=False,  # Worker 内部不需要流式
                ),
                timeout=agent.limits.timeout_seconds
            )
            
            choice = response.choices[0]
            
            # 情况 A: 模型直接回复文本（结束）
            if choice.finish_reason == "stop":
                return AgentResult(
                    final_text=choice.message.content,
                    tool_calls=all_tool_results,
                    usage=response.usage,
                    iterations=iteration + 1,
                )
            
            # 情况 B: 模型要求调用 Tool
            if choice.finish_reason == "tool_calls":
                messages.append(choice.message)  # 加入 assistant 消息
                
                for tool_call in choice.message.tool_calls:
                    tool_name = tool_call.function.name
                    tool_args = json.loads(tool_call.function.arguments)
                    
                    log.info(f"[{trace_id}] Tool调用: {tool_name}({tool_args})")
                    
                    # 执行 Tool
                    tool_result = await self.tool_executor.execute(
                        tool_name, tool_args, context={
                            "agent_id": agent.agent_id,
                            "session_id": task["session_id"],
                            "user_id": task["user_id"],
                            "trace_id": trace_id,
                        }
                    )
                    
                    all_tool_results.append({
                        "tool": tool_name,
                        "args": tool_args,
                        "result": tool_result,
                    })
                    
                    # 把 Tool 结果加入消息
                    messages.append({
                        "role": "tool",
                        "tool_call_id": tool_call.id,
                        "content": json.dumps(tool_result, ensure_ascii=False),
                    })
                
                continue  # 继续下一轮
        
        # 超过最大轮次
        return AgentResult(
            final_text="处理步骤过多，已停止。请简化您的请求。",
            tool_calls=all_tool_results,
            usage=response.usage,
            iterations=max_iterations,
        )
    

    async def publish_reply(self, task: dict, result: AgentResult):
        """将回复发布到 Redis，由 Reply Dispatcher 投递"""
        
        reply = {
            "type": "reply",
            "task_id": task["task_id"],
            "trace_id": task["trace_id"],
            "reply_channel": task["reply_channel"],
            "reply_chat_id": task["reply_chat_id"],
            "reply_message_id": task["reply_message_id"],
            "reply_app_id": task["reply_app_id"],
            "reply_tenant_id": task["reply_tenant_id"],
            "chat_type": task["chat_type"],
            "text": result.final_text,
            "usage": json.dumps({
                "input_tokens": result.usage.prompt_tokens,
                "output_tokens": result.usage.completion_tokens,
                "total_tokens": result.usage.total_tokens,
            }),
            "completed_at": datetime.now().isoformat(),
        }
        
        await self.redis.xadd("reply:outbox", reply, maxlen=100000)
```

```python
# dispatcher/reply_dispatcher.py

class ReplyDispatcher:
    """
    监听 Redis reply:outbox 队列
    根据 reply_channel 调用对应渠道 API 发送回复
    """
    
    def __init__(self, redis_client):
        self.redis = redis_client
        
        # 各渠道回复客户端
        self.channels = {
            "dingtalk": DingtalkReplyClient(),
            "wecom": WecomReplyClient(),
            "feishu": FeishuReplyClient(),
        }
    
    async def start(self):
        try:
            await self.redis.xgroup_create("reply:outbox", "dispatchers", 
                                           id="0", mkstream=True)
        except Exception:
            pass
        
        while True:
            messages = await self.redis.xreadgroup(
                "dispatchers", "dispatcher-1",
                {"reply:outbox": ">"},
                count=10,
                block=1000
            )
            
            for stream, entries in messages:
                for msg_id, reply_data in entries:
                    try:
                        await self.dispatch(reply_data)
                        await self.redis.xack("reply:outbox", "dispatchers", msg_id)
                    except Exception as e:
                        log.error(f"回复投递失败: {e}")
                        # 失败不 ACK，会被重新消费（at-least-once）
    
    async def dispatch(self, reply: dict):
        channel = reply["reply_channel"]
        client = self.channels.get(channel)
        if not client:
            log.error(f"未知渠道: {channel}")
            return
        
        await client.send_reply(reply)
```
